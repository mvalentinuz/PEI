212 void Analizador::algoritmoDelPintor(int fila, int columna, Pixel referencia, int tolerancia)
213 {
214  ++profundidadRecursiva;
215  if(profundidadRecursiva<3000){
216  if(fila>=0 && fila<punteroImagen->getFilas() && columna<punteroImagen->getColumnas() && columna>=0)
217  {
218  Pixel pixel = punteroImagen->getPixel(fila, columna);
219  int deltaR = sqrt(pow( (pixel.getR() - referencia.getR()) , 2));

221  int deltaB = sqrt(pow( (pixel.getB() - referencia.getB()) , 2));
222 
223  if(deltaR<=tolerancia && deltaG<=tolerancia && deltaB<=tolerancia && !region[fila][columna])
224  {

226  areaMapeada++;
227  algoritmoDelPintor(fila+1, columna+1, referencia, tolerancia);
